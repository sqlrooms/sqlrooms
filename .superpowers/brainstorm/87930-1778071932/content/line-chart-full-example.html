LineChartSettings: Full Example

Complete implementation with explicit component composition

Schema (existing)

// packages/mosaic/src/chart-types/line-chart/schema.ts
export
type
LineChartSettings = {
x?: string;
xInterval?: TemporalInterval;
yFields: YFieldConfig[]; // { field, aggregate?, color? }[]
}

Option A: Direct Props (Simple)

// packages/mosaic/src/chart-types/line-chart/LineChartSettings.tsx

interface LineChartSettingsProps {
columns: ChartBuilderColumn[];
values: LineChartSettings;
onChange: (values: LineChartSettings) => void;
}

export function LineChartSettings({
columns,
values,
onChange
}: LineChartSettingsProps) {
// Helper to update a single field
const updateField = useCallback(
(key: keyof LineChartSettings, value: any) => {
onChange({...values, [key]: value});
},
[values, onChange]
);

return (
<div className="space-y-4">
{/* X Axis with optional temporal granularity */}
<FieldSelector
label="X Axis"
required
>
<ColumnSelector
columns={columns}
types={['DATE', 'TIMESTAMP', 'INTEGER', 'DOUBLE']}
value={values.x}
onChange={(x) => updateField('x', x)}
/>
{values.x && <TemporalGranularitySelector
value={values.xInterval}
onChange={(xInterval) => updateField('xInterval', xInterval)}
/>}
</FieldSelector>

{/* Y Axis with multiple fields */}
<MultiFieldSelector
label="Y Axes"
required
>
{values.yFields.map((field, index) => (
<MultiFieldSelector.Row key={field.field}>
<ColumnSelector
columns={columns}
types={['INTEGER', 'DOUBLE', 'BIGINT']}
value={field.field}
onChange={(newField) => {
const updated = [...values.yFields];
updated[index] = {...field, field: newField};
updateField('yFields', updated);
}}
/>
{values.xInterval && <AggregationSelector
value={field.aggregate || 'sum'}
onChange={(agg) => {
const updated = [...values.yFields];
updated[index] = {...field, aggregate: agg};
updateField('yFields', updated);
}}
/>}
<DeleteButton
onClick={() => {
updateField('yFields', values.yFields.filter((_, i) => i !== index));
}}
/>
</MultiFieldSelector.Row>
))}
<MultiFieldSelector.AddRow
columns={columns}
types={['INTEGER', 'DOUBLE', 'BIGINT']}
onAdd={(field) => {
updateField('yFields', [...values.yFields, {field, aggregate: 'sum'}]);
}}
/>
</MultiFieldSelector>
</div>
);
}

Pros

  • Explicit data flow - easy to understand
  • Full control over composition
  • No magic context
  • Chart-specific logic clear

Cons

  • Verbose onChange handlers for arrays
  • Repetitive updateField calls

Option B: Compound Component with Context

// packages/mosaic/src/chart-types/line-chart/LineChartSettings.tsx

export function LineChartSettings({
columns,
values,
onChange
}: LineChartSettingsProps) {
return (
<ChartSettingsProvider value={{columns, values, onChange}}>
<div className="space-y-4">
<LineChartSettings.XAxis />
<LineChartSettings.YAxis />
</div>
</ChartSettingsProvider>
);
}

// Internal subcomponents

const {columns, values, updateField} = useChartSettings<LineChartSettings>();

return (

value={values.xInterval}

}

const {columns, values, updateField} = useChartSettings<LineChartSettings>();

return (
<MultiFieldSelector label="Y Axes" required>
{values.yFields.map((field, index) => (
// Same as Option A but updateField from context
))}

}

Pros

  • Cleaner component organization
  • No prop drilling
  • Easy to split into files

Cons

  • Context setup overhead
  • Less explicit data flow
  • Still verbose array updates

Key Decision: MultiFieldSelector State Management

The trickiest part is managing the yFields array. Each row needs to update its own field/aggregate/color without recreating the array on every render.

Should MultiFieldSelector own the array state internally?

  • Yes: Simpler API - pass value and onChange for the whole array
  • No: Parent component explicitly manages each field update
