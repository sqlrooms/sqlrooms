Integration in ChartSettingsPanel

// packages/mosaic/src/dashboard/chart-settings/ChartSettingsPanel.tsx

import {LineChartSettings} from '../../chart-types/line-chart';

function ChartSettingsPanel({chartConfig, onChange, ...}) {
// Check if chart type has explicit settings component
const renderSettings = () => {
if (chartConfig.chartType === 'line-chart') {
return (
<LineChartSettings
columns={columns}
values={chartConfig.settings}
onChange={(settings) => onChange({...chartConfig, settings})}
/>
);
}

// Fallback to DynamicChartSettings for other chart types
return (
<DynamicChartSettings
chartTypeDefinition={getChartTypeDefinition(chartConfig.chartType)}
columns={columns}
values={chartConfig.settings}
onChange={(settings) => onChange({...chartConfig, settings})}
/>
);
};

return /* ... render settings ... */;
}

Migration Strategy

  • Create all new components in chart-builders/
  • Create LineChartSettings.tsx using new components
  • Update ChartSettingsPanel to check for explicit settings
  • Test thoroughly with line-chart
  • Once proven, migrate other chart types one by one
  • Eventually remove DynamicChartSettings when all charts migrated
